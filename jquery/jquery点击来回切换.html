    方法二：

    <div class="people_state">设为管理员</div>

    <div class="people_state">设为管理员</div>

    <js>

    $(".people_state").each(function() {

          $(".people_state").bind("click");

          $(".people_state").toggle(function() {

          $(this).html("取消管理员");

          },function() {

         $(this).html("设为管理员");

     })

    });

    方法四：

    <html>

    <div class="people_state"    res="0">设为管理员</div>

     

     

    <div class="people_state"    res="0">设为管理员</div>

     

    <js>

    $(".people_state").click(function() {

        var  i=$(this).attr("res");

        if(i==0){

               $(this).attr("res","1");

               $(this).html("");

               $(this).html("取消管理员");

        }else{

              $(this).attr("res","0");

              $(this).html("");

              $(this).html("设为管理员");

        }

    });
