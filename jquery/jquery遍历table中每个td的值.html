    废话不说直接上：

    <table style="border:1px solid #ccc">
            <tr>
                <td>q</td>
                <td>w</td>
                <td>e</td>
                <td>r</td>
            </tr>
            <tr>
                <td>d</td>
                <td>f</td>
                <td>2</td>
                <td>连</td>
            </tr>

        </table>

    先记得导入jquery.js

    $('table tr').each(function(){

        var txt = $(this).find('td:eq(3)').text();  //这里选中的是每个tr里面的第4个td

            console.log(txt)
        })
