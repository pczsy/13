    2、采用 datagrid-groupview 分组展示后，取消分组展示。

    　datagrid使用分组展示 是基于view属性，把view属性设置为groupview 即可分组展示。取消分组时，需要把view属性设置为DefaultView，但是easyui本身没有暴露 DefaultView的变量。所以需要在界面load的时候，存储datagrid的默认view然后，在取消分组操作时给view变量赋值。

    var gridDefaultView;
//表单加载后事件
$(function () {
    //存储默认的视图，用于冲掉分组视图
    gridDefaultView = $('#dgJGQuery').datagrid('options').view;
});

//分组合计
function ShowGroupView(data) {
        $('#dgJGQuery').datagrid({
            groupField: groupCol,
            view: groupview,
            groupFormatter: function (value, rows) {
                return value;
            }
        });
}
