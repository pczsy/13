    1. API使用
     
    存储数据
    1、不指定cookie过期时间
    $.session.set('key', 'value')
     
    2、设置cookie过期时间
    $.session.set('key', 'value', true)
     
    注：默认过期时间为2周
     
     
    获取数据
    $.session.get('key');
     
    删除数据
    $.session.remove('key');
     
    清空所有数据
    $.session.clear();
