        if (!partn.exec(orginnumber))

            let number = [...orginnumber]; //es6语法，利用...将变量的值转成数组。

            let ration22= new Array("1","0","X","9","8","7","6","5","4","3","2");
            let sum=0;
            console.log(ration);
                for(let i=0;i<number.length-1;i++){

                        console.log(i);
