    • select input 等控件进行清空操作

    <html>
<head>
    <meta charset="utf-8" />
    <title></title>
    <script src="http://code.jquery.com/jquery-2.1.0.js"></script>
    <script type="text/javascript">

    
        function reset() {
    　　　　　　　　//清除文本框内容  初始化
    　　　　　　　　//这种方法对select同样适用
    　　　　　　　　//自定义控件也可以适用同样的方法进行初始化
    　　　　　　　　//这种清空是清空内容或者是选中项目 清空之后提交 value=null
            $(    '#txt').val(null).trigger("change");
        }
    </script>
</head>
<body>
    <input type="text" id="txt" />
    <button onclick="reset()">清空</button>
</body>
