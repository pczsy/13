    Js代码!--------

    $.confirm({
            title: "标题", //false 去掉标题
            content: '确认删除该信息？',
            confirm: function () {//点击确认 ajax请求后台
　　　　　　　　　　$.ajax({
                    type: 'POST',
                    url: "@Request.RawUrl",
                    dataType: "json",
                    data: { guid: '@guid', op: 'deleteData', UUID: item.UUID }
               　　 }).done(function (response) {
                    if (!response.Success) {}
    　　　　　　　　　　　　　　　　});
 　　　　　　},
            cancel:     function () {//点击取消不做任何操作
            }
    })
