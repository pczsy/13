        // 方式3.有个动画隐藏和显示,
        $(".wrap>ul>li").on("click", function () {
            var next = $(this).children(".child-ul");
            var icon = $(this).children("span");
            next.slideToggle('fade');
            if (icon.html() === "+") {
                icon.html("-");
            } else {
                icon.html("+");
            }
            $('.child-ul').not(next).slideUp('fast'); //不是当前点击的内容全部向上收起
            $('.wrap>ul>li').children("span").not(icon).html("+");
            return false;
        })

    })
