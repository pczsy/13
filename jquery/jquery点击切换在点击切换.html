    <div  style=" position: absolute;   top: 6rem;   right: 4rem;" id="hides" >
        <a class="icon_a" style="color: #333; font-size: 16px;display: flex;"><span title=".icon32  .icon-color  .icon-square-plus " class="icon32 icon-color icon-square-plus"></span>展开表格
        </a>

        <a class="icon_a2"  style="color: #333; font-size: 16px;display: none;">
            <span title=".icon32 .icon-color .icon-square-minus " class="icon32 icon-color icon-square-minus"></span>收起表格
        </a>

    $(document).ready(function(){
        setTimeout(function () {
            $("#hides").toggle(function(){
                    $(".icon_a").css("display","none");
                    $(".icon_a2").css("display","flex");
                    $("#settleAuditTb").slideToggle("slow");
                },
                function(){
                    $(".icon_a").css("display","flex");
                    $(".icon_a2").css("display","none");
                    $("#settleAuditTb").slideToggle("hidden");
                }
            );
        },1000)
    });
