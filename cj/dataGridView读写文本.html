    FileStream fs = new FileStream(".//allitems.txt", System.IO.FileMode.Open, System.IO.FileAccess.Read);

    StreamReader sr = new StreamReader(fs, System.Text.Encoding.GetEncoding("utf-8"));
    try
    {
    while (true)
    {
    z = sr.ReadLine();
    if (z == null) break;
    dataGridView1.Rows.Add(z);

    }
