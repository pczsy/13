    DataTable去除重复行

    利用DataView的ToTable()方法，方法第一个参数代表是否去除重复，true则去除，第二个参数传列名。

    //新建datatable 实际上应该是其他函数返回来的 这里演示所以新建
DataTable data = new DataTable();
//添加列
data.Columns.Add(new DataColumn("subject"));
//添加行
data.Rows.Add("aa");
data.Rows.Add("aa");
data.Rows.Add("bb");
data.Rows.Add("cc");

//创建DataView
DataView dv = data.DefaultView;
    
    //调用DataRow的ToTable方法
DataTable newData = dv.ToTable(true, "subject");
