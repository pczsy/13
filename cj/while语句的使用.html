    学习while语句的正确用法：
    题目：老师教学生，讲一次之后 问学生会不会，如果不会；就再讲一遍。如果会了就放学，但是如果连讲了十遍还是不会，那也要放学回家

    namespace _44讲十遍会不会
{
    class Program
    {
        /// <summary>
        /// 此语句主要是对信息录入进行计数，如果在一定次数内无法录入正确数据，那么系统会退出
        /// 常用于系统登录过程中，防止不停的录入用户名或密码。
        /// </summary>
        /// <param name="args"></param>
        static void Main(string[] args)
        {
            string answer = "";
            int i = 0;

            //Console.WriteLine("这是我第{}次讲给你听，你现在会不会",i+1);

           // Console.ReadKey();

            while (answer != "yes" && i < 10)
            {
                Console.WriteLine("这是我第{0}次讲给你听，你现在会不会",i+1);
                answer = Console.ReadLine();

                if (answer == "yes")
                {
                    Console.WriteLine("现在放学！！");
                    break;
                }
                i++;
            }
        }
    }
}
