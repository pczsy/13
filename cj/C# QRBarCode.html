    using IronBarCode; 
 class Program
    {         
        static void Main(string[] args)
        {
            BarCodeDemo();
            Console.ReadLine();
        }

        static void BarCodeDemo()
        {
            string barCodeMsg = "Be bold and make a difference!";
            GeneratedBarcode MyBarCode =BarcodeWriter.CreateBarcode(barCodeMsg, BarcodeWriterEncoding.QRCode);
            string pictureFilePath = DateTime.Now.ToString("yyyyMMddHHmmssFFFF") + ".jpg";
            MyBarCode.SaveAsPng(pictureFilePath);
            System.Diagnostics.Process.Start(pictureFilePath);                       
        }
}
